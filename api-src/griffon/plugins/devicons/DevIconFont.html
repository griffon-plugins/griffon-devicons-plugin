218     private static final String ERROR_DESCRIPTION_BLANK = "Argument 'description' must not be blank";
219 
220     public static DevIconFont findByDescription(@Nonnull String description) {
221         requireNonBlank(description, "Icon description must not be blank.");
222 
223         String[] parts = description.split(":");
224         for (DevIconFont font : values()) {
225             if (font.getDescription().equals(parts[0])) {
226                 return font;
227             }
228         }
229 
230         throw new IllegalArgumentException("Icon description '" + description + "' is invalid!");
231     }
232 
233     private String description;
234     private char code;
235 
236     DevIconFont(@Nonnull String description, char code) {
237         this.description = description;
238         this.code = code;
239     }
240 
241     @Nonnull
242     public String getDescription() {
243         return description;
244     }
245 
246     public char getCode() {
247         return code;
248     }
249 
250     @Nonnull
251     public static IllegalArgumentException invalidDescription(@Nonnull String description) {
252         requireNonBlank(description, ERROR_DESCRIPTION_BLANK);
253         throw new IllegalArgumentException("Description " + description + " is not a valid DevIcon icon description");
254     }
255 
256     @Nonnull
257     public static IllegalArgumentException invalidDescription(@Nonnull String description, Exception e) {
258         requireNonBlank(description, ERROR_DESCRIPTION_BLANK);
259         throw new IllegalArgumentException("Description " + description + " is not a valid DevIcon icon description", e);
260     }
